# to_afg

    usage: to_afg -i <reads file> [arguments ...]

    arguments:
        -i, --reads <file>
            (required)
            input fasta/fastq reads file
        --fastq
            default: fasta format
            format of input reads file
        -o, --out <file>
            output afg reads file
        -h, -help
            prints out the help
